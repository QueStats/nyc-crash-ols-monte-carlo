# This is the workflow to reproduce the results of the nyc-crash-injury-bootstrap-vs-monte-carlo
# project by Quincy Cornish.

# The raw data file must be located at:
# ~/Downloads/Motor_Vehicle_Collisions_-_Crashes.csv

# Step 1: Run the bootstrap study.
# The first argument is the random number seed.
# The second argument is the number of bootstrap samples.

for seed in {1..1}
do
    Rscript bootstrap.r $seed 10000
done

# Step 2: Produce all figures and tables from the saved bootstrap output.

Rscript out_file.r 1

# Step 3: Compile the report after the figures and tables are created.
